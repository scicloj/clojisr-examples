Most basic violin chart

(plot->file
   (str target-path "a.png")
   (let [data (base/data-frame
                :name [(repeat 500 "A") (repeat 500 "B")
                       (repeat 500 "B") (repeat 20 "C")
                       (repeat 100 "D")]
                :value [(stats/rnorm 500 10 5) (stats/rnorm 500 13 1)
                        (stats/rnorm 500 18 1) (stats/rnorm 20 25 4)
                        (stats/rnorm 100 12 1)])]
     (r+ (gg/ggplot data (gg/aes :x 'name :y 'value :fill 'name))
         (gg/geom_violin))))
